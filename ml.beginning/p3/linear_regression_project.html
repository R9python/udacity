                c = A[i][j]

def gj_Solve(A, b, decPts=4, epsilon = 1.0e-16):
    row_A,col_A = shape(A)
    row_B,col_B = shape(b)

    if(row_A == row_B):
        MA = augmentMatrix(A,b)
        row,col = shape(MA)
        for j in range(col-1):
            maxValue, maxValueRow = djx_max(MA,j)
            if maxValue <= epsilon: #code:<=,<
                # 0,MA,None 
                return None 
            else:
                # step1:c
                swapRows(MA, maxValueRow,j) #,xy,j

                # step2:c1
                scale = 1.0/MA[j][j] #code:1.0
                scaleRow(MA,j,scale)

                # step3:c0
                for i in range(row):
                    if i != j:
                        scale2 = MA[i][j] * -1.0 #code:1.0
                        addScaledRow(MA,i,j,scale2)

        # MA
        MT = transpose(MA)
        row_mt,_ = shape(MT)
        return [[round(x,decPts)]for x in MT[row_mt-1]] #code:
    else:
        return None

def calculateMSE(X,Y,m,b):
    total = 0
    point_count = len(X)
    for i in range(point_count):
        total += pow(Y[i] - X[i]*m - b,2)
    return total/point_count
